Log Transform → Pairplot → Correlation HeatmapCode
pract4_multivariate.pyPython 
# Log transformation to reduce skew
def log_transform(data, col):
    for colname in col:
        if (data[colname] <= 1.0).any():
            data[colname + '_log'] = np.log(data[colname] + 1)
        else:
            data[colname + '_log'] = np.log(data[colname])

log_transform(data, ['friend_count', 'likes'])

# Distribution of log-transformed friend_count
plt.figure(figsize=(8, 5))
sns.distplot(data["friend_count_log"], axlabel="friend_count_log")
plt.show()

# Bivariate pairplot (sample 1000 rows for speed)
sns.pairplot(data=data.drop(['friend_count', 'likes'], axis=1).sample(1000))
plt.show()

# Multivariate correlation heatmap
plt.figure(figsize=(12, 7))
sns.heatmap(
    data.drop(['friend_count', 'likes'], axis=1).corr(numeric_only=True),
    annot=True, vmin=-1, vmax=1
)
plt.show()

# 5000 negative tweets (target=0)

df_pos = pd.read_csv(file_path, encoding="latin-1",

df = pd.concat([df_neg, df_pos], ignore_index=True)

Assign Sentiment Labels & Calculate PolarityCode
pract5_sentiment.pyPython 
# Label: 0 = Negative, 4 = Positive
df["sentiment"] = df["target"].apply(
    lambda x: "Negative" if x == 0 else "Positive"
)

# TextBlob polarity: >0 positive, 0 neutral, <0 negative
def get_polarity(text):
    return TextBlob(str(text)).sentiment.polarity

comments_url = (

comments_response = requests.get(comments_url)

def get_comment_sentiment(comment):
    analysis = TextBlob(comment)
    if analysis.sentiment.polarity > 0:
        return "Positive"
    elif analysis.sentiment.polarity == 0:
        return "neutral"
    else:
        return "negative"
